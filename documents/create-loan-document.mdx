---
title: "Create a loan document"
description: "Creates a new document and associates it with the specified loan"
api: "POST /loan/{id}/document"
---


## Path Parameters

<ParamField path="id" type="string" required>
  The loan ID or loan number
</ParamField>

## Request Body

<ParamField body="Name" type="string" required>
  Document name
</ParamField>

<ParamField body="Description" type="string">
  Document description
</ParamField>

<ParamField body="Status" type="string">
  Document status (e.g., pending, approved)
</ParamField>

<ParamField body="Summary" type="string">
  Document summary
</ParamField>

<ParamField body="Is_Shared" type="boolean">
  Whether the document is shared with the borrower
</ParamField>

<ParamField body="Uploaded_Date" type="string">
  ISO 8601 date when the file was uploaded
</ParamField>

<ParamField body="Requested_Date" type="string">
  ISO 8601 date when the document was requested
</ParamField>

<ParamField body="Expired_Date" type="string">
  ISO 8601 date when the document expires
</ParamField>

<ParamField body="Shared_Date" type="string">
  ISO 8601 date when the document was shared
</ParamField>

<ParamField body="Borrower_Id" type="string">
  ID of the borrower to associate
</ParamField>

<ParamField body="Guarantor_Id" type="string">
  ID of the guarantor to associate
</ParamField>

<ParamField body="Fund_Id" type="string">
  ID of the fund to associate
</ParamField>

<ParamField body="Order" type="integer">
  Display order
</ParamField>

<ParamField body="Section" type="string">
  Section the document belongs to
</ParamField>

<ParamField body="Instructions" type="string">
  Instructions for the document
</ParamField>
